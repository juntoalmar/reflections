COMMANDS

git clone URL  	-- copia repositorio a local
git log  		-- muestra historial de log
git diff commit1 commit2 -- muestra diferencias entre commits
git checkout		-- restaurar una versión anterior
git init		-- crea un repository en un directorio (sin ningún commit)
git status		-- da informacion del repository
git commit 		-- hace un commit del directorio
git add	<file>	-- añade un fichero al "staging area" para hacer commit
git reset		-- remove file from staging area
